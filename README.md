        plt.plot(x, -12*x**4*np.sin(np.cos(x)) - 18*x**3+5*x**2 + 10*x - 30,

        plt.xlabel(r'$x$', fontsize=12)
        plt.ylabel(r'$f(x)$', fontsize=12)
        plt.grid(True)
        plt.legend(loc='best', fontsize=12)
        plt.savefig('figure.png')
        plt.show()
